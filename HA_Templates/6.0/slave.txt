end

edit port1
set mode static
set ip 192.168.1.12 255.255.255.0
set allowaccess https ping ssh fgfm
set alias public
next
edit port2
set mode static
set ip 192.168.2.12 255.255.255.0
set allowaccess ping
set alias private
next
edit port3
set mode static
set ip 192.168.3.12 255.255.255.0
set allowaccess ping
set alias hasync
next
edit port4
set mode static
set ip 192.168.4.12 255.255.255.0
set allowaccess https ping ssh
set alias hamgmt
next
end
config router static
edit 1
set device port1
set gateway 192.168.1.1
next
edit 2
set device port2
set dst 192.168.0.0/16
set gateway 192.168.2.1
next
end
config system dns

end
config firewall policy
    edit 1
        set name "outbound-all"
        set srcintf "port2"
        set dstintf "port1"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ALL"
        set logtraffic all
        set nat enable
    next
end

    set group-name "group1"

    set session-pickup enable

    config ha-mgmt-interface 
        edit 1
            set interface port4
            set gateway 192.168.4.1
        next
    end
    set override disable
    set priority 1
    set unicast-hb enable
